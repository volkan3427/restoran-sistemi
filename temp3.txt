async function ortakMasalariCek() {
    if (!firebaseUrl) return;
    // Eğer bu cihaz masalara az önce sipariş girdiyse (son 3 saniye), 
    // sunucudan eski verinin gelip ekranı geri sarmasını (siparişin kaybolup gelmesini) önlemek için atla:
    if (window._sonMasaYukleme && (Date.now() - window._sonMasaYukleme < 5000)) return;
    try {
      const t = Date.now();
      const r = await fetch(`${firebaseUrl}/v3_ortak_masalar.json?_t=${t}`, { cache: 'no-store' });
      if (!r.ok) return;
      const data = await r.json();
      if (!data || typeof data !== 'object') return;

      // Firebase boş dizileri (adisyon: []) ve null'ları veritabanından siler. 
      // Çekilen veride bozulan yapıyı tamir edelim:
      masaNoları.forEach(no => {
        if (!data[no]) data[no] = {};
        if (!data[no].adisyon) data[no].adisyon = [];
      });

      const yeniJson = JSON.stringify(data);
      const mevcutJson = JSON.stringify(masalar);

      if (yeniJson === mevcutJson) { _sonOrtakMasalarDegisikligi = yeniJson; return; }
      if (yeniJson !== mevcutJson && yeniJson !== _sonOrtakMasalarDegisikligi) {
        if (seciliMasa && data[seciliMasa]) {
            data[seciliMasa].adisyon = Object.assign([], masalar[seciliMasa].adisyon);
            data[seciliMasa].acilisZamani = masalar[seciliMasa].acilisZamani;
            data[seciliMasa].isim = masalar[seciliMasa].isim;
        }
        masalar = data;
        _sonOrtakMasalarDegisikligi = yeniJson;
        
        // Sadece local'e kaydet (firebase'e geri basmayı önlemek için)
        try { localStorage.setItem('restoran_masalar', JSON.stringify(masalar)); } catch(e) {}
        
        // Ekrana yansıt
        masaRender();
        if (seciliMasa) panelRender();
      }
    } catch(e) {}
  }
